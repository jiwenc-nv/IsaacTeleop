# Do not pin Linux arm64 to nlopt==2.6.2 here: that forces dex-retargeting 0.4.x
# and NumPy 1.x, which cannot coexist with Pinocchio 3.x in combined environments.
nlopt>=2.6.2

# Pinocchio for JointStateRetargeter(mode="ee_pose"): URDF forward kinematics for
# joint-space leader arms / exoskeletons. Declared explicitly (rather than relying
# on dex-retargeting pulling it transitively) so EE-mode is reproducible.
# `mode="joint"` never imports it.
#
# Constraints MUST mirror requirements-grounding.txt: CI installs every extra into
# one environment, so a higher `pin` floor here (e.g. pin>=4, which needs
# cmeel-urdfdom>=6) is unsatisfiable against grounding's cmeel-urdfdom<5. The cmeel
# caps keep pin's compiled bindings linking the sonames they were built against;
# newer cmeel-urdfdom 6.x / cmeel-tinyxml2 11.x break `import pinocchio` at runtime.
pin>=2.7.0
cmeel-urdfdom<5
cmeel-tinyxml2<11
